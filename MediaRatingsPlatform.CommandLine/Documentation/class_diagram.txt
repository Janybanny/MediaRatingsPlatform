    class AuthAPI {

        +Register() : bool

        -String Password
    }
    
    class UserAPI {

        +EditProfile()
        +DisplayRatings()
        +DisplayFavourites()
    }
    
    class MediaAPI {
        +ListMedia(filters)
        +DisplayMedia(int mediaId)

        +UpdateMedia(int mediaId, struct MediaInput) : int status
    }
    
    class RatingAPI {

package BusinessLayer {

    class Verification {
        +Verification(String token)
        +CheckAccess() : Boolean
        +CheckUserAccess(int userId) : Boolean
        +CheckMediaAccess(int mediaId) : Boolean
        +CheckRatingAccess(int ratingId) : Boolean
    }
    Verification --> TokenHandler
    
    class TokenHandler {
        +addToken(int userId)
        +checkToken(int token)
        +invalidateToken(int token)
    }
    
    interface IDataRequest {
        +getData(int id)
    }

    class GetProfile {
        +getData(int id) : json
    }
    IDataRequest <|.. GetProfile

    class GetRatings {
        +getData(int id) : json
    }
    IDataRequest <|.. GetRatings

    class GetFavourites {
        +getData(int id) : json
    }
    IDataRequest <|.. GetFavourites

    class ListMedia {
        +getData(int id) : json
        +getFilteredData(int id, filter) : json
    }
    IDataRequest <|.. ListMedia

    class GetMedia {
        +getData(int id) : json
    }
    IDataRequest <|.. GetMedia

    class RecommendationsEngine {
        +getData(int id) : json
    }
    IDataRequest <|.. RecommendationsEngine

    class GetLeaderboard {
        +getData(int id) : json
    }
    IDataRequest <|.. GetLeaderboard

    
    interface IUserManipulation {
        +createUser()
        +editUser(int userId)
    }
    
    interface IMediaManipulation {
        +createMedia()
        +editMedia(int mediaId)
        +deleteMedia(int mediaId)
        +addFavourite(int userId, mediaId)
    }
    
    interface IRatingManipulation {
        +createRating()
        +editRating(int ratingId)
        +deleteRating(int ratingId)
        +likeRating(int userId, int ratingId)
        +confirmComment(int ratingId)
    }
    
}
